package com.example.app

import android.os.Bundle
import android.widget.*
import androidx.appcompat.app.AppCompatActivity

class MainActivity : AppCompatActivity() {

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_main)

        val w = findViewById<EditText>(R.id.weight)
        val h = findViewById<EditText>(R.id.height)
        val btn = findViewById<Button>(R.id.calc)
        val res = findViewById<TextView>(R.id.res)

        btn.setOnClickListener {
            val weight = w.text.toString().toFloat()
            val height = h.text.toString().toFloat()

            val bmi = weight / (height * height)

            res.text = "BMI = $bmi"
        }
    }
}

package com.example.app

import android.os.Bundle
import android.widget.*
import androidx.appcompat.app.AppCompatActivity

class MainActivity : AppCompatActivity() {

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_main)

        val name = findViewById<EditText>(R.id.name)
        val email = findViewById<EditText>(R.id.email)
        val pass = findViewById<EditText>(R.id.pass)
        val btn = findViewById<Button>(R.id.register)

        btn.setOnClickListener {

            if (name.text.isEmpty())
                name.error = "Enter Name"

            else if (email.text.isEmpty())
                email.error = "Enter Email"

            else if (pass.text.isEmpty())
                pass.error = "Enter Password"

            else
                Toast.makeText(this, "Registered", Toast.LENGTH_SHORT).show()
        }
    }
}

package com.example.app

import android.content.Intent
import android.net.Uri
import android.os.Bundle
import android.widget.Button
import androidx.appcompat.app.AppCompatActivity

class MainActivity : AppCompatActivity() {

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_main)

        val call = findViewById<Button>(R.id.call)
        val web = findViewById<Button>(R.id.web)

        call.setOnClickListener {
            val i = Intent(Intent.ACTION_DIAL)
            startActivity(i)
        }

        web.setOnClickListener {
            val i = Intent(Intent.ACTION_VIEW,
                Uri.parse("https://www.google.com"))
            startActivity(i)
        }
    }
}

package com.example.app

import android.content.Context
import android.os.Bundle
import android.widget.*
import androidx.appcompat.app.AppCompatActivity

class MainActivity : AppCompatActivity() {

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_main)

        val name = findViewById<EditText>(R.id.name)
        val save = findViewById<Button>(R.id.save)

        val sp = getSharedPreferences("data", Context.MODE_PRIVATE)

        name.setText(sp.getString("username", ""))

        save.setOnClickListener {

            val editor = sp.edit()
            editor.putString("username", name.text.toString())
            editor.apply()

            Toast.makeText(this, "Saved", Toast.LENGTH_SHORT).show()
        }
    }
}
